通用 Makefile 文件#

############### Makefile Common ###############
# C or C++
EXT = c
# gcc or g++
CC = gcc
# includes directory
INCDIR = includes
# sources directory
SRCDIR = sources
# target directory
TARDIR = target
# .o file directroy
OBJDIR = $(TARDIR)/obj
# .d file directory
DEPDIR = $(TARDIR)/dep
# exec filename
EXEC = $(TARDIR)/main.exe

# 预处理选项
CPPFLAGS += -I$(INCDIR)

# link
LDFLAGS +=

#  编译器选项
CFLAGS += -O -std=c11 -g -Wall

SRCS = $(foreach dir, $(SRCDIR), $(wildcard $(dir)/*.$(EXT)))
OBJS = $(patsubst %.$(EXT), $(OBJDIR)/%.o, $(notdir $(SRCS)))
DEPS = $(patsubst %.$(EXT), $(DEPDIR)/%.d, $(notdir $(SRCS)))

all: $(EXEC)
	@echo Successful

$(EXEC): $(OBJS)
	@$(CC) $^ -o $@

ifneq "$(MAKECMDGOALS)" "clean"
sinclude $(DEPS)
endif

$(OBJDIR)/%.o: $(SRCDIR)/%.$(EXT)
	@$(CC) $(CFLAGS) $(CPPFLAGS) -c $< -o $@

$(DEPDIR)/%.d: $(SRCDIR)/%.$(EXT)
	@$(CC) $(CFLAGS) $(CPPFLAGS) -MM -MF $@ $< -MT $(DEPDIR)/$*.d -MT $(OBJDIR)/$*.o

.PHONY: all clean

clean:
	@del /Q /S $(TARDIR)
